def add(*args)
  args.each do |value|
    if @head == nil then
      @head = Node.new(value, nil, nil)
      @lastnext = @head
    else
      last = Node.new(value, nil, @lastnext)
      last.father = @lastnext
      @lastnext.next = last
      @lastnext = last
    end
  end
end

def first

    aux = @head.value
    @head = @head.next
    return aux
  else
    return -1
  end

def reverse

def to_a
  a_value = Array.new
  pointer = @head
  while pointer.next != nil do
    a_value.push(pointer.value)
    pointer = pointer.next
  end
  a_value.push(pointer.value)
  a_value.to_s
end
